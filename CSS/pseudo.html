Anker-Pseudoklassen:

:link • :visited • :hover • :active

/*--Anker-Pseudoklassen für Link-Elemente--*/

/*CSS*/  
a:link { color: #f8b500; 
         font-size: 26px; 
         text-decoration: none; }
  
a:visited { color: #37a34a; 
            font-size: 26px; 
            text-decoration: none; }
 
a:hover { color: #9f6f55; 
          font-size: 26px; 
          text-decoration: none; 
          font-weight: 900; }

a:active { color: #ffec47; 
           font-size: 26px; 
           text-decoration: none; 
           font-weight: 900; }
        
    <!--HTML-->
    <a href="https://www.google.de/" 
    target="_blank">Link nach Google</a>
        

/*----:hover auf div-Elemente anwenden----*/

/*CSS*/ 
.hoverbox { height: 60px;
            width: 280px;
            background-color: #f6b483; 
            margin-top: 30px; }

.hoverbox p { padding: 19px 11px; }

.hoverbox:hover { background-color: #fff462; }
        
   <!--HTML-->
   <div class="hoverbox">
     <p>Bitte mit der Maus über mich fahren</p>
   </div>
