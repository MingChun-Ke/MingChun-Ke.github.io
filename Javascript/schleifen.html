Schleifen:

for • for...in • while • do...while

for

Syntax:

for (var Zählvariable = Startwert;
            Zählvariable <= Endwert;
            Zählvariable erhöhen/vermindern) {
     Anweisungen; //wird durch eine bestimmte Wiederholungszahl ausgeführt.
}
Beispiel
    //JavaScript
    function forFun() {
       var x = "";
       for (var y = 0; y <= 5; y++) {
            x = x + "no." + y + " ";
       }
    document.getElementById("fordemo").innerHTML = x;
    }
        
    //HTML
    <button onclick="forFun()">hier klicken</button>
    <span id="fordemo"></span>

for...in

Syntax:

for (var Eigenschaften in Objekt / var Elememte in Array) {
     Anweisungen; /*wird alle Eigenschaften eines Objekts oder
                             alle Elemente eines Arrays durchlaufen.*/
}
Beispiel
    //JavaScript
    function forInFun() {
       var y = "";
       var dogNames = ["Lucy", "Coco", "Jack"];
       var x;
       for (x in dogNames) {
            y = y + x + ":" + dogNames[x] + " ";
       }
    document.getElementById("forindemo").innerHTML = y;
    }
        
    //HTML
    <button onclick="forInFun()">hier klicken</button>
    <span id="forindemo"></span>
Output:

while

Syntax:

while(Bedingung) {
   Anweisungen; /*wird ausgeführt, solange die Bedingung wahr ist.
             (wenn die erste Bedingung nicht wahr ist, wird diese Schleife gar nicht erst ausgeführt)*/
}
Beispiel
    //JavaScript
    function whileFun() {
       var x = "";
       var y = 0;
       while (y <= 5) {
            x = x + "no." + y + " ";
            y++;
       }
    document.getElementById("whiledemo").innerHTML = x;
    }
        
    //HTML
    <button onclick="whileFun()">hier klicken</button>
    <span id="whiledemo"></span>
Output:

do...while

Syntax:

do {
    Anweisungen; /*wird einmal ausgeführt, bevor die Bedingung geprüft wird.
                            Wenn die Bedingung wahr ist, wird die Schleife ein weiteres Mal durchlaufen.*/
} while (Bedingung);
Beispiel
    //JavaScript
    function doWhileFunction() {
       var x = "";
       var y = 0;
       do {
           x = x + "no." + y + " ";
           y++;
       } while ( y <= 5);
    document.getElementById("dowhiledemo").innerHTML = x;
    }
        
    //HTML
    <button onclick="doWhileFunction()">hier klicken</button>
    <span id="dowhiledemo"></span>
