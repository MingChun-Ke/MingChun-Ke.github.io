Bedingte Anweisungen:

if • if...else • if...else if...else • switch

if

Syntax:

if (Bedingung) {
   Anweisungen; //wird ausgeführt, wenn die Bedingung wahr ist.
}
Beispiel
    //JavaScript
    function ifFun() {
       var x = "";
       var time = new Date();
       var weekDay = time.getDay();
       if (weekDay > 0) {
           x="Heute ist ein Werktag.";
       }
    document.getElementById("ifdemo").innerHTML = x;
    }

    //HTML
    <button onclick="ifFun()">hier klicken</button>
    <span id="ifdemo"></span>
Output:

if...else

Syntax:

if (Bedingung) {
   Anweisungen1; //wird ausgeführt, wenn die Bedingung wahr ist.
} else {
   Anweisungen2; //wird ausgeführt, wenn die Bedingung falsch ist.
}
Beispiel
    //JavaScript
    function ifElseFun() {
       var x = "";
       var time = new Date().getHours();
       if (time < 12) {
           x = "Jetzt ist AM (vor dem Mittag)";
       } else {
           x = "Jetzt ist PM (nach dem Mittag)";
       }
    document.getElementById("ifelsedemo").innerHTML = x;
    }

    //HTML
    <button onclick="ifElseFun()">hier klicken</button> 
    <span id="ifelsedemo"></span>
Output:

if...else if...else

Syntax:

if (Bedingung1) {
   Anweisungen1; //wird ausgeführt, wenn die Bedingung1 wahr ist.
} else if (Bedingung2) {
   Anweisungen2; //wird ausgeführt, wenn die Bedingung1 falsch und die Bedingung2 wahr sind.
} else {
   Anweisungen3; //wird ausgeführt, wenn die Bedingung1 und die Bedingung2 falsch sind.
}
Beispiel
    //JavaScript
    function ifElseIfElseFun() {
       var x="";
       var time=new Date().getHours();
       if (time>=6 && time<10) {
               x = "Guten Morgen!";
       } else if (time>=10 && time<17) {
               x = "Guten Tag!";
       } else if (time>=17 && time<23) {
               x = "Guten Abend!";
       } else {
               x = "Gute Nacht!";
       }
    document.getElementById("ifelseifelsedemo").innerHTML = x;
    }

    //HTML
    <button onclick="ifElseIfElseFun()">hier klicken</button> 
    <span id="ifelseifelsedemo"></span>
Output:

switch

Syntax:

switch (Variable oder Ausdruck) {
    case Wert1:  
        Anweisungen1; //wird ausgeführt, falls Ausdruck oder Variable mit Wert1 übereinstimmt.
        break;
    case Wert2:  
        Anweisungen2; //wird ausgeführt, falls Ausdruck oder Variable mit Wert2 übereinstimmt.
        break;
         
    ......
         
    default:  
        Anweisungen; //wird ausgeführt, falls keiner der anderen Fälle erfüllt ist.
        break;
}
Beispiel
    //JavaScript
    function switchFun() {
       var day;
       switch (new Date().getDay()) {
          case 1:
              day = "Heute ist Montag...";
              break;
          case 2:
              day = "Heute ist Dienstag...";
              break;
          case 3:
              day = "Heute ist Mittwoch...";
              break;
          case 4:
              day = "Heute ist Donnerstag...";
              break;
          case 5:
              day = "Heute ist Freitag!";
              break;
          default:
              day = "Wochenende!";
       }
    document.getElementById("switchdemo").innerHTML = day;
    }

    //HTML
    <button onclick="switchFun()">hier klicken</button> 
    <span id="switchdemo"></span>
